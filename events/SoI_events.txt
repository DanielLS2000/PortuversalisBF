namespace = SoI

# Embassy under Siege
country_event = {
	id = SoI.1
	title = SoI.1.t
	desc = SoI.1.d
	picture = COUNTRY_COLLAPSE_eventPicture
	
	is_triggered_only = yes
	
	option = { # We must protect our interests, with force.
		name = SoI.1.ra
		add_casus_belli = {
			target = FROM
			type = cb_restore_sphere
			months = 12
		}
		if = {
			limit = {
				ai = yes
				is_great_power_trigger = yes
				is_at_war = no
			}
			if = {
				limit = {
					military_strength = {
						who = FROM
						value = 3.0
					}
				}
				declare_war_with_cb = {
					who = FROM
					casus_belli = cb_restore_sphere
				}
			}
			else_if = {
				limit = {
					military_strength = {
						who = FROM
						value = 2.0
					}
				}
				random = {
					chance = 75
					declare_war_with_cb = {
						who = FROM
						casus_belli = cb_restore_sphere
					}
				}
			}
			else_if = {
				limit = {
					military_strength = {
						who = FROM
						value = 1.0
					}
				}
				random = {
					chance = 50
					declare_war_with_cb = {
						who = FROM
						casus_belli = cb_restore_sphere
					}
				}
			}
		}
	}
	
	option = { # We'll salvage what little influence we have left.
		name = SoI.1.rb
		add_prestige = -15
		add_diplomatic_influence = {
			who = FROM
			value = 10
		}
	}
}
