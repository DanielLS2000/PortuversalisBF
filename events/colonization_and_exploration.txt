# Settler Pool Calculations #
country_event = {
	id = colonial_events.14
	title = colonial_events.14_t
	desc = colonial_events.14_d

	hidden = yes

	is_triggered_only = yes

	immediate = {

		export_to_variable = {
			which = numProvincesInStates
			value = trigger_value:num_of_provinces_in_states
		}
		multiply_variable = {
			which = numProvincesInStates	# provinces in states * 30
			value = 30
		}
		export_to_variable = {
			which = totalDevelopment
			value = trigger_value:total_development
		}
		multiply_variable = {
			which = totalDevelopment	# total development * 2
			value = 2
		}
		change_variable = {
			which = numProvincesInStates
			which = totalDevelopment
		}
		change_variable = {
			which = numProvincesInStates
			value = 1000				# base value
		}
		# Decision add 1000 settlers
		if = {
			limit = {
				has_country_flag = decisionAddSettlers_flag
			}
			change_variable = {
				which = numProvincesInStates
				value = 500
			}
		}
		if = {
			limit = {
				had_country_flag = {
					flag = decisionAddSettlers_flag
					days = 1825
				}
			}
			clr_country_flag = decisionAddSettlers_flag
			country_event = {
				id = colonial_events.161
			}
		}

		set_variable = {
			which = maxNumOfSettlers	# maximum number of settlers of a country
			which = numProvincesInStates
		}
		# AI Bonus
		export_to_variable = {
			which = numOfColonists	# current number of colonies
			value = trigger_value:num_of_colonists
		}
		if = {
			limit = {
				ai = yes		# ai gets 500 extra settlers per each colonist
			}
			multiply_variable = {
				which = numOfColonists
				value = 500
			}
			change_variable = {
				which = maxNumOfSettlers
				which = numOfColonists
			}
		}

		if = {
			limit = {
				has_idea = siberian_frontier
			}
			change_variable = {
				which = maxNumOfSettlers
				value = 3000
			}
		}

		export_to_variable = {
			which = numOfColonies	# current number of colonies
			value = trigger_value:num_of_colonies
		}
		set_variable = {
			which = usedSettlers
			which = numOfColonies
		}
		multiply_variable = {
			which = usedSettlers	# settlers currently colonizing
			value = 1000
		}

		set_variable = {
			which = proxyMaxNumOfSettlers
			which = maxNumOfSettlers
		}
		subtract_variable = {
			which = proxyMaxNumOfSettlers
			which = usedSettlers
		}
		set_variable = {
			which = freeSettlers	# currently available settlers
			which = proxyMaxNumOfSettlers
		}
	}

	option = {
		name = colonial_events.14a

		set_variable = {
			which = 0settlers	# 0% of max settlers
			value = 0			
		}
		change_variable = {
			which = 0settlers
			value = 1
		}

		set_variable = {
			which = 25settlers
			which = maxNumOfSettlers
		}
		multiply_variable = {
			which = 25settlers	# 25% of max settlers
			value = 0.25
		}

		set_variable = {
			which = 50settlers
			which = maxNumOfSettlers
		}
		multiply_variable = {
			which = 50settlers	# 50% of max settlers
			value = 0.50
		}

		set_variable = {
			which = 75settlers
			which = maxNumOfSettlers
		}
		multiply_variable = {
			which = 75settlers	# 75% of max settlers
			value = 0.75
		}

		if = {
			limit = {
				NOT = {
					check_variable = {
						which = freeSettlers
						# which = 0settlers
						value = 1
					}
				}
			}
			add_country_modifier = {
				name = no_settlers
				duration = -1
			}
			remove_country_modifier = very_low_settlers
			remove_country_modifier = low_settlers
			remove_country_modifier = some_settlers
			remove_country_modifier = many_settlers
		}
		if = {
			limit = {
				check_variable = {
					which = freeSettlers
					# which = 0settlers
					value = 1
				}
				NOT = {
					check_variable = {
						which = freeSettlers
						# which = 25settlers
						value = 500
					}
				}
			}
			add_country_modifier = {
				name = very_low_settlers
				duration = -1
			}
			remove_country_modifier = no_settlers
			remove_country_modifier = low_settlers
			remove_country_modifier = some_settlers
			remove_country_modifier = many_settlers
		}
		if = {
			limit = {
				AND = {
					check_variable = {
						which = freeSettlers
						# which = 25settlers
						value = 500
					}
					NOT = {
						check_variable = {
							which = freeSettlers
							# which = 50settlers
							value = 1000
						}
					}
				}
			}
			add_country_modifier = {
				name = low_settlers
				duration = -1
			}
			remove_country_modifier = no_settlers
			remove_country_modifier = very_low_settlers
			remove_country_modifier = some_settlers
			remove_country_modifier = many_settlers
		}
		if = {
			limit = {
				AND = {
					check_variable = {
						which = freeSettlers
						# which = 50settlers
						value = 1000
					}
					NOT = {
						check_variable = {
							which = freeSettlers
							# which = 75settlers
							value = 2000
						}
					}
				}
			}
			add_country_modifier = {
				name = some_settlers
				duration = -1
			}
			remove_country_modifier = no_settlers
			remove_country_modifier = very_low_settlers
			remove_country_modifier = low_settlers
			remove_country_modifier = many_settlers
		}
		if = {
			limit = {
				check_variable = {
					which = freeSettlers
					# which = 75settlers
					value = 2000
				}
			}
			add_country_modifier = {
				name = many_settlers
				duration = -1
			}
			remove_country_modifier = no_settlers
			remove_country_modifier = very_low_settlers
			remove_country_modifier = low_settlers
			remove_country_modifier = some_settlers
		}
	}
}

# Christopher Columbus
country_event = {
	id = colonial_events.2
	title = "colonial_events.2.t"
	desc = "colonial_events.2.d"
	picture = SHIP_SAILING_eventPicture
	
	trigger = {
		NOT = { has_global_flag = columbus_happened }
		NOT = { has_country_flag = turned_down_columbus }
		full_idea_group = exploration_ideas
		is_year = 1476
		OR = {
			NOT = {
				north_america = {
					has_discovered = ROOT
				}
			}
			NOT = {
				south_america = {
					has_discovered = ROOT
				}
			}
			NOT = {
				any_known_country = {
					north_america = {
						has_discovered = PREV
					}
				}
			}
			NOT = {
				any_known_country = {
					south_america = {
						has_discovered = PREV
					}
				}
			}
		}
	}

	mean_time_to_happen = {
		months = 120
		modifier = {
			factor = 0.2
			is_year = 1485
		}
		modifier = {
			factor = 0.5
			AND = {
				capital_scope = {
					region = iberia_region
				}
				is_year = 1485
			}
		}
	}

	option = {		# Reject his proposal
		name = "colonial_events.2a"
		ai_chance = { factor = 10 }
		set_country_flag = turned_down_columbus
		random_known_country = {
			limit = {
				NOT = { has_country_flag = turned_down_columbus }
				religion = catholic
				has_idea = quest_for_the_new_world
			}
			country_event = { id = colonial_events.2 }
		}
	}
	option = {		# Hire Columbus
		name = "colonial_events.2b"
		ai_chance = { factor = 90 }
		add_years_of_income = -0.25
		set_global_flag = columbus_happened
		hidden_effect = {
			ROOT = {
				country_event = {
					id = colonial_events.17
					days = 90
				}
			}
		}
	}
}
